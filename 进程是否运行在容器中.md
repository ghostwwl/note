#!/bin/bash
# 通用容器检测脚本

is_container() {
    # 方法1：检查容器标识文件
    if [ -f /.dockerenv ] || [ -f /.podmanenv ] || [ -f /.containerenv ] || [ -f /.lxc_env ]; then
        return 0
    fi

    # 方法2：检查cgroup路径
    if grep -q -E "(docker|libpod|lxc|rkt|kubelet|containerd)" /proc/self/cgroup 2>/dev/null; then
        return 0
    fi

    # 方法3：对比命名空间
    if [ -r /proc/1/ns/pid ] && [ -r /proc/self/ns/pid ]; then
        host_ns=$(readlink /proc/1/ns/pid)
        self_ns=$(readlink /proc/self/ns/pid)
        [ "$host_ns" != "$self_ns" ] && return 0
    fi

    # 方法4：检查容器运行时进程
    if pgrep -x "dockerd" >/dev/null 2>&1 || pgrep -x "containerd" >/dev/null 2>&1 || pgrep -x "podman" >/dev/null 2>&1 || pgrep -x "lxc" >/dev/null 2>&1 || pgrep -x "rkt" >/dev/null 2>&1; then
        return 0
    fi

    # 方法5：检查环境变量
    if [ -n "${KUBERNETES_SERVICE_HOST}" ] || [ -n "${CONTAINER}" ] || [ -n "${DOCKER_HOST}" ]; then
        return 0
    fi

    return 1
}

detect_container_type() {
    # 检测具体容器类型
    if grep -q docker /proc/self/cgroup 2>/dev/null || [ -f /.dockerenv ] || [ -n "${DOCKER_HOST}" ]; then
        echo "容器类型: Docker"
    elif grep -q libpod /proc/self/cgroup 2>/dev/null || [ -f /.podmanenv ] || [ -f /.containerenv ]; then
        echo "容器类型: Podman"
    elif grep -q lxc /proc/self/cgroup 2>/dev/null || [ -f /.lxc_env ]; then
        echo "容器类型: LXC"
    elif grep -q rkt /proc/self/cgroup 2>/dev/null; then
        echo "容器类型: rkt"
    elif [ -n "${KUBERNETES_SERVICE_HOST}" ]; then
        echo "容器类型: Kubernetes"
    elif grep -q containerd /proc/self/cgroup 2>/dev/null; then
        echo "容器类型: Containerd"
    else
        echo "容器类型: 未知"
    fi
}

if is_container; then
    echo "当前运行在容器环境中"
    detect_container_type
else
    echo "当前运行在宿主机环境中"
fi
